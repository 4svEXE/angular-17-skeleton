## Add Angular
    yarn add @angular/cli
## Create Angular app 
    ng new app-name --skip-install
    cd ./app-name
    npm i  //loooong -> yarn install
## Runn server
    ng serve
## To open
    http://localhost:4200
## Add tailwind
    yarn add -D tailwindcss postcss autoprefixer
    npx tailwindcss init

	@tailwind components;
	@tailwind utilities;
